terraform {
  required_providers {
    vault = {
      source  = "hashicorp/vault"
      version = "~> 4.0"
    }
  }
}

provider "vault" {}


data "vault_kv_secret_v2" "db" {
  mount = "secret"
  name  = "db"
}


output "username" {
  value = data.vault_kv_secret_v2.db.data["username"]
}

output "password" {
  value     = data.vault_kv_secret_v2.db.data["password"]
  sensitive = true
}
